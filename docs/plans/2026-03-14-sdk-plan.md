# SDK Build System Wrapper — Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Create a general-purpose SDK (`sdk/`) for compiling C/C++ programs targeting the wasm-posix-kernel, with autoconf support.

**Architecture:** Node.js TypeScript package using native type stripping (`--experimental-strip-types`). Thin wrappers around LLVM tools that inject wasm32-specific flags. Shared library modules handle toolchain discovery, flag filtering, and child process execution.

**Tech Stack:** TypeScript, Node.js 22+, vitest (tests), LLVM/clang (wrapped)

**Design doc:** `docs/plans/2026-03-14-sdk-design.md`

import { execFile, spawn } from 'node:child_process';

export interface RunResult {
  stdout: string;
  stderr: string;
  exitCode: number;
}

/**
 * Run a command and capture its output. Does not throw on non-zero exit.
 */
export function run(cmd: string, args: string[]): Promise<RunResult> {
  return new Promise((resolve) => {
    execFile(cmd, args, { maxBuffer: 10 * 1024 * 1024 }, (error, stdout, stderr) => {
      resolve({
        stdout: stdout ?? '',
        stderr: stderr ?? '',
        exitCode: error ? (error as any).code ?? 1 : 0,
      });
    });
  });
}

/**
 * Run a command with stdio inherited (passthrough to terminal).
 * Returns the exit code.
 */
export function runPassthrough(cmd: string, args: string[], env?: NodeJS.ProcessEnv): Promise<number> {
  return new Promise((resolve) => {
    const child = spawn(cmd, args, {
      stdio: 'inherit',
      env: env ?? process.env,
    });
    child.on('close', (code) => resolve(code ?? 1));
    child.on('error', () => resolve(127));
  });
}
```

import { existsSync, readdirSync } from 'node:fs';
import { join, resolve, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';
import { run } from './exec.ts';

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);

/** SDK root is sdk/, repo root is one level up */
const SDK_ROOT = resolve(__dirname, '../..');
const REPO_ROOT = resolve(SDK_ROOT, '..');

export interface Toolchain {
  llvmDir: string;
  cc: string;
  cxx: string;
  ar: string;
  ranlib: string;
  nm: string;
  sysroot: string;
  glueDir: string;
}

const HOMEBREW_LLVM = '/opt/homebrew/opt/llvm/bin';
const LINUX_LLVM_PREFIX = '/usr/lib/llvm-';

/**
 * Find a working LLVM directory with wasm32 support.
 */
export async function findLlvmDir(): Promise<string> {
  // 1. Env var override
  const envDir = process.env.WASM_POSIX_LLVM_DIR;
  if (envDir) {
    if (existsSync(join(envDir, 'clang'))) return envDir;
    throw new Error(`WASM_POSIX_LLVM_DIR="${envDir}" does not contain clang`);
  }

  // 2. clang on PATH — check if it supports wasm32
  const pathResult = await run('which', ['clang']);
  if (pathResult.exitCode === 0) {
    const clangPath = pathResult.stdout.trim();
    const testResult = await run(clangPath, ['--target=wasm32-unknown-unknown', '-x', 'c', '-c', '-o', '/dev/null', '-', '--']);
    if (testResult.exitCode === 0) {
      return dirname(clangPath);
    }
  }

  // 3. Homebrew (macOS)
  if (existsSync(join(HOMEBREW_LLVM, 'clang'))) return HOMEBREW_LLVM;

  // 4. Linux system LLVM (pick highest version)
  try {
    const parent = '/usr/lib';
    const entries = readdirSync(parent).filter(e => e.startsWith('llvm-')).sort();
    for (const entry of entries.reverse()) {
      const binDir = join(parent, entry, 'bin');
      if (existsSync(join(binDir, 'clang'))) return binDir;
    }
  } catch {
    // /usr/lib may not exist or not be readable
  }

  throw new Error(
    'Could not find a wasm32-capable LLVM/clang installation.\n' +
    'Install LLVM via:\n' +
    '  macOS:  brew install llvm\n' +
    '  Ubuntu: apt install llvm clang lld\n' +
    'Or set WASM_POSIX_LLVM_DIR to your LLVM bin directory.'
  );
}

/**
 * Find the musl sysroot.
 */
export function findSysroot(): string {
  const envSysroot = process.env.WASM_POSIX_SYSROOT;
  if (envSysroot) return envSysroot;
  return resolve(REPO_ROOT, 'sysroot');
}

/**
 * Validate that the sysroot has been built.
 */
export function validateSysroot(sysroot: string): void {
  if (!existsSync(join(sysroot, 'lib', 'libc.a'))) {
    throw new Error(
      `Sysroot not found at ${sysroot}\n` +
      'Run scripts/build-musl.sh first.'
    );
  }
}

/**
 * Find the glue directory.
 */
export function findGlueDir(): string {
  const envGlue = process.env.WASM_POSIX_GLUE_DIR;
  if (envGlue) return envGlue;
  return resolve(REPO_ROOT, 'glue');
}

/**
 * Resolve the full toolchain.
 */
export async function resolveToolchain(): Promise<Toolchain> {
  const llvmDir = await findLlvmDir();
  const sysroot = findSysroot();
  const glueDir = findGlueDir();

  validateSysroot(sysroot);

  return {
    llvmDir,
    cc: join(llvmDir, 'clang'),
    cxx: join(llvmDir, 'clang++'),
    ar: join(llvmDir, 'llvm-ar'),
    ranlib: join(llvmDir, 'llvm-ranlib'),
    nm: join(llvmDir, 'llvm-nm'),
    sysroot,
    glueDir,
  };
}
```

/** Flags always injected during compilation. */
export const COMPILE_FLAGS: string[] = [
  '--target=wasm32-unknown-unknown',
  '-matomics',
  '-mbulk-memory',
  '-fno-exceptions',
  '-fno-trapping-math',
];

/** Flags injected during linking (in addition to COMPILE_FLAGS). */
export const LINK_FLAGS: string[] = [
  '-nostdlib',
  '-Wl,--entry=_start',
  '-Wl,--export=_start',
  '-Wl,--import-memory',
  '-Wl,--shared-memory',
  '-Wl,--max-memory=1073741824',
  '-Wl,--allow-undefined',
];

/** Flags silently dropped (no warning). */
const IGNORED_EXACT = new Set([
  '-pthread',
  '-lpthread',
  '-fPIC',
  '-fPIE',
  '-pie',
  '-ldl',
  '-lrt',
  '-lresolv',
  '-lm',
  '-lcrypt',
  '-lutil',
  '-rdynamic',
  '-Wl,-Bsymbolic',
]);

/** Prefixes of flags silently dropped. */
const IGNORED_PREFIXES = [
  '-Wl,-rpath,',
  '-Wl,-soname,',
];

/** Flags that produce a warning. */
const WARN_FLAGS = new Set([
  '-shared',
  '-dynamiclib',
]);

export interface FilterResult {
  filtered: string[];
  warnings: string[];
}

/**
 * Filter an argument list: remove ignored/warned flags, pass through the rest.
 */
export function filterArgs(args: string[]): FilterResult {
  const filtered: string[] = [];
  const warnings: string[] = [];

  for (const arg of args) {
    if (IGNORED_EXACT.has(arg)) continue;
    if (IGNORED_PREFIXES.some(p => arg.startsWith(p))) continue;
    if (WARN_FLAGS.has(arg)) {
      warnings.push(`wasm32posix-cc: warning: ${arg} is not supported for Wasm targets (ignored)`);
      continue;
    }
    filtered.push(arg);
  }

  return { filtered, warnings };
}

export interface ParsedArgs {
  compileOnly: boolean;    // -c present
  preprocessOnly: boolean; // -E present
  assemblyOnly: boolean;   // -S present
  outputFile: string | null;
  sourceFiles: string[];   // .c, .cc, .cpp, .cxx, .m files
  objectFiles: string[];   // .o files
  archiveFiles: string[];  // .a files
  otherArgs: string[];     // everything else
}

const SOURCE_EXTS = new Set(['.c', '.cc', '.cpp', '.cxx', '.m', '.mm', '.i', '.ii']);
const OBJECT_EXTS = new Set(['.o']);
const ARCHIVE_EXTS = new Set(['.a']);

/**
 * Parse arguments to determine compilation mode and categorize inputs.
 */
export function parseArgs(args: string[]): ParsedArgs {
  const result: ParsedArgs = {
    compileOnly: false,
    preprocessOnly: false,
    assemblyOnly: false,
    outputFile: null,
    sourceFiles: [],
    objectFiles: [],
    archiveFiles: [],
    otherArgs: [],
  };

  for (let i = 0; i < args.length; i++) {
    const arg = args[i];
    if (arg === '-c') {
      result.compileOnly = true;
    } else if (arg === '-E') {
      result.preprocessOnly = true;
    } else if (arg === '-S') {
      result.assemblyOnly = true;
    } else if (arg === '-o') {
      i++;
      result.outputFile = args[i] ?? null;
    } else if (!arg.startsWith('-')) {
      const ext = arg.substring(arg.lastIndexOf('.'));
      if (SOURCE_EXTS.has(ext)) {
        result.sourceFiles.push(arg);
      } else if (OBJECT_EXTS.has(ext)) {
        result.objectFiles.push(arg);
      } else if (ARCHIVE_EXTS.has(ext)) {
        result.archiveFiles.push(arg);
      } else {
        result.otherArgs.push(arg);
      }
    } else {
      result.otherArgs.push(arg);
    }
  }

  return result;
}

/**
 * Determine if this invocation needs linking.
 * Linking happens when there are inputs and no -c/-E/-S flag.
 */
export function needsLinking(parsed: ParsedArgs): boolean {
  if (parsed.compileOnly || parsed.preprocessOnly || parsed.assemblyOnly) return false;
  return parsed.sourceFiles.length > 0 || parsed.objectFiles.length > 0;
}
```

#!/usr/bin/env -S node --experimental-strip-types
import { join } from 'node:path';
import { resolveToolchain, type Toolchain } from '../lib/toolchain.ts';
import { COMPILE_FLAGS, LINK_FLAGS, filterArgs, parseArgs, needsLinking } from '../lib/flags.ts';
import { runPassthrough } from '../lib/exec.ts';

/**
 * Build the full clang argument list based on the user's args and the toolchain.
 * Exported for testing.
 */
export function buildClangArgs(userArgs: string[], toolchain: Toolchain): string[] {
  const { filtered, warnings } = filterArgs(userArgs);
  for (const w of warnings) console.error(w);

  const parsed = parseArgs(filtered);

  const args: string[] = [
    ...COMPILE_FLAGS,
    `--sysroot=${toolchain.sysroot}`,
  ];

  // Reassemble non-file args
  if (parsed.compileOnly) args.push('-c');
  if (parsed.preprocessOnly) args.push('-E');
  if (parsed.assemblyOnly) args.push('-S');
  if (parsed.outputFile) args.push('-o', parsed.outputFile);
  args.push(...parsed.otherArgs);

  // Add source and object files
  args.push(...parsed.sourceFiles);
  args.push(...parsed.objectFiles);
  args.push(...parsed.archiveFiles);

  // Add linking extras when needed
  if (needsLinking(parsed)) {
    args.push(
      join(toolchain.glueDir, 'syscall_glue.c'),
      join(toolchain.glueDir, 'compiler_rt.c'),
      join(toolchain.sysroot, 'lib', 'crt1.o'),
      join(toolchain.sysroot, 'lib', 'libc.a'),
      ...LINK_FLAGS,
    );
  }

  return args;
}

async function main(): Promise<void> {
  const toolchain = await resolveToolchain();
  const args = buildClangArgs(process.argv.slice(2), toolchain);
  const exitCode = await runPassthrough(toolchain.cc, args);
  process.exit(exitCode);
}

// Only run main when executed directly (not imported for testing)
const isMain = process.argv[1] && import.meta.url.endsWith(process.argv[1].replace(/\\/g, '/'));
if (isMain) main();
```

#!/usr/bin/env -S node --experimental-strip-types
import { runPassthrough } from '../lib/exec.ts';

/**
 * Build the environment variables for ./configure.
 */
export function buildConfigureEnv(): Record<string, string> {
  return {
    ...process.env as Record<string, string>,
    CC: 'wasm32posix-cc',
    CXX: 'wasm32posix-c++',
    AR: 'wasm32posix-ar',
    RANLIB: 'wasm32posix-ranlib',
    NM: 'wasm32posix-nm',
    STRIP: 'wasm32posix-strip',
    PKG_CONFIG: 'wasm32posix-pkg-config',
    CFLAGS: process.env.CFLAGS ?? '',
    CXXFLAGS: process.env.CXXFLAGS ?? '',
    LDFLAGS: process.env.LDFLAGS ?? '',
  };
}

/**
 * Build the argument list for ./configure.
 */
export function buildConfigureArgs(userArgs: string[]): string[] {
  return [
    '--host=wasm32-unknown-none',
    '--prefix=/usr',
    ...userArgs,
  ];
}

async function main(): Promise<void> {

  const env = buildConfigureEnv();
  const exitCode = await runPassthrough('./configure', args, env);
  process.exit(exitCode);
